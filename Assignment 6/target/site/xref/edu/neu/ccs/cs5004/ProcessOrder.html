32    /**
33     * Method helps check and remove any underage items from a shoppers cart. If the customer is buying
34     * alcohol, it checks their current age. If the current age is below 21, then the alcohol is removed
35     * from their cart.
36     */
37    public void removeUnderAgeProducts() {
38      if (currentCart.getShoppingCart().containsKey(beer) && customer.getAge() < 21) {
39        currentCart.removeItemFromCart(beer);
40      }
41    }
42  
43    /**
44     * This method helps update the stock quantities after we process an order. Once a cart purchase
45     * successfully goes through, it updates the stock with the new current amounts by subtracting the
46     * item purchased from the exisiting supply.
47     *
48     * @param cart the shoppers shopping cart
49     */
50    public void updateStockQuantities(ShoppingCart cart) {
51      for (Products products : cart.getShoppingCart().keySet()) {
52        Integer currentProductQuantity = stockItem.getQuantity(products);
53        stockItem.reduceInventory(products, currentProductQuantity);
54      }
55    }
56  
57    /**
58     * This method generates a list of strings that acts as a receipt. For each item in the final cart
59     * it gets the product's name followed by a colon and the price.
60     *
61     * @param cart the shoppers shopping cart
62     * @return a receipt, represented as a list of strings
63     */
64    public List<String> generateReceipt(ShoppingCart cart) {
65      List<String> receipt = new ArrayList<>();
66      for (Products products : cart.getShoppingCart().keySet()) {
67        receipt.add(products.getProductName() + " : " + products.getPrice());
68      }
69      return receipt;
70    }
71  
72    /**
73     * This method removes all products from the cart. When a shopper has succesfully finished their
74     * shopping spree, and purchased all the items, this function goes through their cart 1 by 1 removing
75     * each product.
76     *
77     * @param cart the shoppers shopping cart
78     * @return an empty shopping cart
79     */
80    public ShoppingCart emptyCart(ShoppingCart cart) {
81      for (Products products : cart.getShoppingCart().keySet()) {
82        cart.removeItemFromCart(products);
83      }
84      return cart;
85    }
86  }
