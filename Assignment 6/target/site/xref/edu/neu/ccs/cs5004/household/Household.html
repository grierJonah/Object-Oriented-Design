5   /**
6    * A class for all Household products.
7    */
8   public class Household extends Products {
9   
10    private Integer units;
11  
12    /**
13     * Constructs a household product given the parameters.
14     *
15     * @param manufacturerName the name of the manufacturer
16     * @param productName the name of the product
17     * @param price the cost of the household product, represented as a Double
18     * @param minimumAge the minimum age to buy the product
19     * @param units the number of units each household product sells in
20     */
21    public Household(String manufacturerName, String productName,
22        Double price, Integer minimumAge, Integer units) {
23      super(manufacturerName, productName, price, minimumAge);
24      this.units = units;
25    }
26  
27    /**
28     * Gets the quantity each household product is sold in (i.e. PaperTowels could have 2 in each
29     * sleeve).
30     *
31     * @return the number of units within one purchase
32     */
33    public Integer getUnits() {
34      return units;
35    }
36  }
