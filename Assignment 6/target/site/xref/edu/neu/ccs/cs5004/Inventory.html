4   /**
5    * An inventory system that keeps track of a GroceryStock, and a HouseholdStock. These two stocks
6    * keep track of all the products that are revolving in and out of the SuperMarket's system. It also
7    * helps separate the grocery items from the household items by using two different HashMaps. Lastly,
8    * the Inventory class gives the SuperMarket a cleaner and more organized way of accessing total
9    * inventory levels as well as adding new items.
10   */
11  public class Inventory {
12  
13    private StockItem groceryStock;
14    private StockItem householdStock;
15  
16    /**
17     * Constructs a new Inventory.
18     */
19    public Inventory() {
20      groceryStock = new StockItem();
21      householdStock = new StockItem();
22    }
23  
24    /**
25     * Method allows a SuperMarket to add a new product into the groceryStock.
26     *
27     * @param groceryProduct the new product being added to the groceryStock
28     * @param quantity the quantity to add alongside the product
29     */
30    public void addGroceryItem(Products groceryProduct, Integer quantity) {
31      groceryStock.addToStock(groceryProduct, quantity);
32    }
33  
34    /**
35     * Method allows a SuperMarket to add a new product into the householdStock.
36     *
37     * @param houseHoldProduct the new product being added to the householdStock
38     * @param quantity the quantity to add alongside the product
39     */
40    public void addHouseholdItem(Products houseHoldProduct, Integer quantity) {
41      householdStock.addToStock(houseHoldProduct, quantity);
42    }
43  
44    /**
45     * Gets the total inventory price that the SuperMarket currently has. Uses helper functions to
46     * iterate through both the grocery and household products.
47     *
48     * @return the total inventory price, represented as a Double
49     */
50    public Double getTotalRetailValue() {
51      Integer groceryQuantity = 0;
52      Integer householdQuantity = 0;
53      Double groceryTotal = 0.0;
54      Double householdTotal = 0.0;
55      Double currentGroceryRetailValue = 0.0;
56      Double currentHouseholdRetailValue = 0.0;
57  
58      for (Products product : groceryStock.getKeys()) {
59        // Gets the value (quantity associated with product in HashMap)
60        groceryQuantity += groceryStock.getQuantity(product);
61        currentGroceryRetailValue += product.getPrice();
62        groceryTotal = currentGroceryRetailValue * groceryQuantity;
63      }
64  
65      for (Products product : householdStock.getKeys()) {
66        // Gets the value (quantity associated with product in HashMap)
67        householdQuantity += householdStock.getQuantity(product);
68        currentHouseholdRetailValue += product.getPrice();
69        householdTotal = currentHouseholdRetailValue * householdQuantity;
70      }
71  
72      return groceryTotal + householdTotal;
73    }
74  }
