3   import java.util.HashMap;
4   import java.util.Set;
5   
6   /**
7    * A class representing a StockItem.
8    */
9   public class StockItem {
10  
11    private HashMap<Products, Integer> stockItemMap;
12  
13    /**
14     * Constructs a new StockItem as a HashMap. Each StockItem consists of a product as the key, along
15     * with a quantity of that product as the value. This allows a company to easily keep track of
16     * stock inventory levels, as the look up times for each product is O(1) time.
17     */
18    public StockItem() {
19      stockItemMap = new HashMap<>();
20    }
21  
22    /**
23     * Adds a product into the stockItemMap if it is not already contained in the map.
24     *
25     * @param product the product to put in the map
26     * @param quantity the quantity to put in the map
27     */
28    public void addToStock(Products product, Integer quantity) {
29      if (!stockItemMap.containsKey(product)) {
30        stockItemMap.put(product, quantity);
31      }
32    }
33  
34    /**
35     * Checks if the product is in the StockItemMap as well as checks if there is a sufficient
36     * quantity of that item to be purchased. If the purchaser enters an amount that is greater than
37     * than current value stored in the HashMap, then the method will return false.
38     *
39     * @param product the product to search for
40     * @param numQuantity the quantity to check the current value against
41     * @return a true or false indicating if the current stock inventory is sufficient or lacking
42     */
43    public Boolean checkStockInventory(Products product, Integer numQuantity) {
44      return (stockItemMap.containsKey(product) && stockItemMap.get(product) >= numQuantity);
45    }
46  
47    /**
48     * Reduces the stock inventory level by the amount specified. This method must be used alongside
49     * the checkStockInventory level otherwise the stockItemMap value could run into a negative value.
50     * Example) If we do not first use 'checkStockInventory' levels, we could reduce Shampoo current
51     * inventory from 0 to -100 if the user wanted to purchase 100 shampoo's.
52     *
53     * @param product the product to reduce the inventory on
54     * @param numQuantity the amount to reduce the product's current quantity by
55     */
56    public void reduceInventory(Products product, Integer numQuantity) {
57      Integer currentStockLevel = stockItemMap.get(product);
58      stockItemMap.put(product, currentStockLevel - numQuantity);
59    }
60  
61    /**
62     * Gets the quantity associated with the key in the stockItemMap. Helps in inventory class.
63     *
64     * @param product the product to search for
65     * @return the quantity
66     */
67    public Integer getQuantity(Products product) {
68      return stockItemMap.get(product);
69    }
70  
71    /**
72     * Gets the size of the stockItemMap keys. Is used mainly in inventory class.
73     *
74     * @return the number of keys in the stockItemMap
75     */
76    public Integer getStock() {
77      return stockItemMap.keySet().size();
78    }
79  
80    /**
81     * Gets the set of all products in the stockItemMap. Is used mainly in inventory class.
82     *
83     * @return the keys in the stockItemMap
84     */
85    public Set<Products> getKeys() {
86      return stockItemMap.keySet();
87    }
88  
89  
90  }
