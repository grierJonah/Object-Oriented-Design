3   import edu.neu.ccs.cs5004.exceptions.AgeRestrictionException;
4   
5   /**
6    * A class for a Grocery product: Beer.
7    */
8   public class Beer extends Grocery {
9   
10    private static final Integer MINIMUM_AGE = 21;
11  
12    /**
13     * Constructs a new Beer object given the parameters.
14     *
15     * @param manufacturerName the name of the manufacturer
16     * @param productName the name of the product
17     * @param price the price of the product
18     * @param minimumAge the minimum age for the product (in this case it is 21 for alcohol)
19     * @param weight the weight of the product in ounces
20     * @throws AgeRestrictionException thrown if the purchaser is less than 21 years old
21     */
22    public Beer(String manufacturerName, String productName, Double price,
23        Integer minimumAge, Double weight) throws AgeRestrictionException {
24      super(manufacturerName, productName, price, minimumAge, weight);
25      checkAge(minimumAge);
26    }
27  
28    /**
29     * Checks that the minimum age is satisfied.
30     *
31     * @param age the age of the purchaser
32     * @return true if the age is greater than or equal to the minimum age (21)
33     * @throws AgeRestrictionException thrown if the the purchaser is not the legal age
34     */
35    private Boolean checkAge(Integer age) throws AgeRestrictionException {
36      if (age >= MINIMUM_AGE) {
37        return true;
38      } else {
39        throw new AgeRestrictionException("To purchase alcohol you must be 21 or older!");
40      }
41    }
42  }
