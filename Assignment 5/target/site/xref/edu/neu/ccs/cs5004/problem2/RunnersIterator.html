6   public class RunnersIterator implements Iterable {
7   
8     private List<Runner> runnerList;
9     private Runner runner;
10    private Integer currentElement = 0;
11  
12    public RunnersIterator(List<Runner> runners) {
13      this.runnerList = runners;
14    }
15  
16    /**
17     * Returns an iterator over elements of type {@code T}.
18     *
19     * @return an Iterator.
20     */
21    @Override
22    public Iterator iterator() {
23      return new Iterator() {
24        @Override
25        public boolean hasNext() {
26          if (runnerList.size() >= currentElement + 1) {
27            return true;
28          }
29          return false;
30        }
31  
32        @Override
33        public Object next() {
34          if (runnerList.size() >= currentElement + 1) {
35            return runnerList.get(currentElement).getFiveKRaceTimes().get(currentElement++);
36          }
37          return null;
38        }
39      };
40    }
41  
42    /**
43     * Gets the qualified runners in a list of runner times. If the runners list has more than 5,
44     * 5K races completed and at least 1 half marathon race time under 100.0 minutes than they are
45     * qualified to be in the next upcoming half marathon race.
46     *
47     * @param runners the runners times
48     * @return a qualified runner based on specifications
49     */
50    public List<Runner> getQualifiedRunners(List<Double> runners) {
51      Double halfMarathonRaceTime = 100.0;
52      List<Runner> newList = new ArrayList<>();
53      if (runners.size() >= 5) {
54        for (Double i : runner.getHalfMarathonTimes()) {
55          if (i < halfMarathonRaceTime) {
56            newList.add(runner);
57          }
58        }
59        return newList;
60      } else {
61        return null;
62      }
63    }
