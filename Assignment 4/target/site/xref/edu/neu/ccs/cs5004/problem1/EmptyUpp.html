3   import edu.neu.ccs.cs5004.problem1.exceptions.InvalidUsernameException;
4   import edu.neu.ccs.cs5004.problem1.exceptions.PairEmptyException;
5   
6   /**
7    * An empty UserPassPair. An empty UserPassPair contains no attributes.
8    */
9   public class EmptyUpp extends AbstractUpp {
10  
11    EmptyUpp emptyUserPassPair;
12    IUpp nonEmptyUpp;
13  
14    public EmptyUpp() {
15    }
16  
17    @Override
18    public EmptyUpp createEmpty() {
19      return emptyUserPassPair = new EmptyUpp();
20    }
21  
22    /**
23     * Checks whether or not the UserPassPair has a given userName, and password. Returns true if the
24     * UserPassPair is empty and returns false if the UserPassPair is (u,p) non empty.
25     *
26     * @return true if the UserPassPair is empty, false if the UserPassPair is non-empty
27     */
28    @Override
29    public Boolean isEmpty() {
30      return true;
31    }
32  
33    /**
34     * Allows the user to add a userName and password to their UserPassPair. If the UserPassPair is
35     * empty, the new userName and password will be accepted and the user will be returned their
36     * UserPassPair.
37     *
38     * @param userName the selected userName, represented as a String.
39     * @param passWord the selected password, represented as a String.
40     * @return a new UserPassPair with the userName and password
41     */
42    @Override
43    public IUpp add(String userName, String passWord) {
44      return nonEmptyUpp = new NonEmptyUpp(userName, passWord);
45    }
46  
47    /**
48     * Allows the user to replace their existing UserPassPair. If the user tries to replace an empty
49     * UserPassPair they will run into a PairEmptyException.
50     *
51     * @param userName the selected userName, represented as a String
52     * @param passWord the selected password, represented as a String
53     * @throws PairEmptyException an exception thrown if the user tries to use the replace function on
54     * an empty UserPassPair
55     */
56    @Override
57    public void replace(String userName, String passWord) throws PairEmptyException {
58      throw new PairEmptyException("Error: Cannot replace a UserPassPair that is already empty");
59    }
60  
61    /**
62     * The delete function allows a user to delete their UserPassPair. If the UserPassPair is empty
63     * then the method will return an empty UserPassPair.
64     *
65     * @param userName the selected userName, represented as a String.
66     * @return a new empty UserPassPair
67     */
68    @Override
69    public IUpp delete(String userName) {
70      return emptyUserPassPair = new EmptyUpp();
71    }
72  
73    /**
74     * Allows the user to get their password to their UserPassPair combination. To get the password,
75     * the user must provide their userName. If the user uses the method on an empty UserPassPair
76     * then an InvalidUsernameException will be thrown.
77     *
78     * @param userName the selected userName, represented as a String
79     * @return the UserPassPair password, represented as a String
80     * @throws InvalidUsernameException an exception thrown if the user enters the wrong userName or
81     * enters a userName on an empty UserPassPair.
82     */
83    @Override
84    public String getPassword(String userName) throws InvalidUsernameException {
85      throw new InvalidUsernameException("Error: Cannot get password from an empty UserPassPair");
86    }
87  }
