69    /**
70     * Depending on attributes assigned to creating a new cleaning service, this method will help give
71     * accurate prices with the size of the property. There is a base rate for each hour of cleaning
72     * to be $80. A SMALL property takes exactly 1 hour, so there property cleaning fee is $80.00. If
73     * it is a medium house it takes 2 hours, so the property cleaning fee will be $160.00. If the
74     * property is a LARGE property, then it will take roughly 4 hours to complete. For a large
75     * property the cost will be $320.00.
76     *
77     * @return the rate that the property owner will pay based on the size of the property
78     */
79    public Double calculateInteriorCleaningPrice() {
80      if (this.propertySize == small) {
81        return BASE_RATE;
82      } else if (this.propertySize == medium) {
83        return (BASE_RATE * TWO_HOURS);
84      } else {
85        return (BASE_RATE * FOUR_HOURS);
86      }
87    }
88  
89    /**
90     * Depending on attributes assigned to creating a new painting service, this method will help
91     * provide accurate prices with the size of the property. There is a base rate per hour of
92     * painting at $80.0. Additionally, painters calculate hours worked in terms of days. One day is
93     * equal to eight hours of working. For SMALL property's painting takes two days to complete. That
94     * means that a property owner will expect to pay for 16 hours of work. Like SMALL property's,
95     * MEDIUM sized property's also take two days to complete. Purchasers will be expected to also pay
96     * for 16 hours of work. For LARGE property's it takes three days to complete. This means
97     * purchasers will be expected to pay for 24 hours worth of work.
98     *
99     * @return the rate that the property owner will pay based on the size of the property
100    */
101   public Double calculateInteriorPaintingPrice() {
102     if (this.propertySize == small) {
103       return (BASE_RATE * TWO_DAYS);
104     } else if (this.propertySize == medium) {
105       return (BASE_RATE * TWO_DAYS);
106     } else {
107       return (BASE_RATE * THREE_DAYS);
108     }
109   }
110 
111   /**
112    * Depending on attributes assigned to creating a new cleaning or painting service. This method
113    * will help provide accurate prices if the property has any pets. There is a 5% increase in final
114    * price if the property owner has 1 or 2 pets. There is a 7% increase in final price if the
115    * property owner has 3 or more pets.
116    *
117    * @return the percentage increase that the property owner will be required to pay if they have
118    * any pets.
119    */
120   public Double calculatePetPrice() {
121     if (numberOfPets >= 1 && numberOfPets <= 2) {
122       return FIVE_PERCENT;
123     } else if (numberOfPets >= 3) {
124       return SEVEN_PERCENT;
125     } else {
126       return NO_COST;
127     }
128   }
