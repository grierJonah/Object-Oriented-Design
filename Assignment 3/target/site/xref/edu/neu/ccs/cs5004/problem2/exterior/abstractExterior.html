7   /**
8    * An abstract class for Exterior services.
9    */
10  public abstract class abstractExterior extends abstractService {
11  
12    public static final Integer TWO_HOURS = 2;
13    public static final Integer FOUR_HOURS = 4;
14    public static final Double FIVE_PERCENT = 0.05;
15    public static final Double NO_COST = 0.0;
16  
17    protected Integer numberOfFloors;
18    protected Integer month;
19    protected PropertySizeEnum small = PropertySizeEnum.SMALL;
20    protected PropertySizeEnum medium = PropertySizeEnum.MEDIUM;
21  
22    private Double exteriorPrice;
23    protected Double finalPrice;
24  
25  
26    /**
27     * Constructs a new abstractExterior with the given parameters.
28     *
29     * @param invoiceID the invoiceID of the exterior service
30     * @param propertyAddress the property address of the exterior service
31     * @param propertySize the property size of the exterior service, represented as an Enum
32     * @param scheduledMonthly a Boolean value checking if the service is monthly or a 1 time service
33     * @param numberOfServices the number of services the property has had, represented as an Integer
34     */
35    public abstractExterior(String invoiceID, String propertyAddress, PropertySizeEnum propertySize,
36        Boolean scheduledMonthly, Integer numberOfServices, Integer numberOfFloors, Integer month) {
37      super(invoiceID, propertyAddress, propertySize, scheduledMonthly, numberOfServices);
38      this.numberOfFloors = numberOfFloors;
39      this.month = month;
40    }
41  
42    /**
43     * Gets the number of floors a property has.
44     *
45     * @return the number of floors a property has.
46     */
47    public Integer getNumberOfFloors() {
48      return numberOfFloors;
49    }
50  
51    /**
52     * Sets the number of floors a property has.
53     *
54     * @param numberOfFloors the number of floors a property has.
55     */
56    public void setNumberOfFloors(Integer numberOfFloors) {
57      this.numberOfFloors = numberOfFloors;
58    }
59  
60    /**
61     * Gets the month of the scheduled exterior cleaning.
62     *
63     * @return the month, represented as an Integer.
64     */
65    public Integer getMonth() {
66      return month;
67    }
68  
69    /**
70     * Sets the month that the exterior cleaning will happen.
71     *
72     * @param month the month that the scheduled servicing will happen, represented as an Integer.
73     */
74    public void setMonth(Integer month) {
75      this.month = month;
76    }
77  
78  
79    @Override
80    public Double calculatePrice() throws TooManyFloorsException {
81      Double withoutDiscountPrice = calculateExteriorServicingPrice();
82      Double discountPrice = calculateDiscountHelper();
83      return finalPrice = withoutDiscountPrice - (withoutDiscountPrice * discountPrice);
84    }
85  
86    /**
87     * Depending on attributes assigned to creating a new exterior service, this method will help give
88     * accurate prices with the size of the property. There is a base rate for each hour of exterior
89     * work to be $80. A SMALL property takes exactly 2 hours, so there property gardening fee is
90     * $160.00. If it is a medium house it also takes 2 hours, so the property cleaning fee will be
91     * $160.00. If the property is a LARGE property, then it will take roughly 4 hours to complete.
92     * For a large property the cost will be $320.00.
93     *
94     * @return the dollar rate that the property owner will pay based on the size of the property.
95     */
96  
97    protected Double calculateExteriorServicingPrice() {
98      if (this.propertySize == small) {
99        return (BASE_RATE * TWO_HOURS);
100     } else if (propertySize == medium) {
101       return (BASE_RATE * TWO_HOURS);
102     } else {
103       return (BASE_RATE * FOUR_HOURS);
104     }
105   }
106 }
