3   import edu.neu.ccs.cs5004.problem1.Name;
4   import java.util.Objects;
5   
6   /**
7    * A class representing a Substitute Teacher. A Substitute Teacher has all the same parameters of
8    * any other teacher with the addition of having a counting ability to keep track of the amount of
9    * days they have worked in a given classroom.
10   */
11  public class substituteTeacher extends abstractSubstituteTeacher {
12  
13    protected Integer numberOfDaysInAGivenClassroom;
14  
15    /**
16     * Constructs a new Substitute Teacher with the given parameters.
17     *
18     * @param name the name of the Substitute Teacher
19     * @param classroom the name of the classroom
20     * @param coteacher the name of the Co Teacher
21     * @param vacationStatus the vacation status of the Substitute Teacher, represented as a Boolean
22     * @param numberOfDaysInAGivenClassroom the number of days a Substitute Teacher has worked in a
23     * given classroom
24     */
25    public substituteTeacher(Name name, String classroom,
26        Name coteacher, Integer numberOfDaysInAGivenClassroom, Boolean vacationStatus) {
27      super(name, classroom, coteacher, numberOfDaysInAGivenClassroom, vacationStatus);
28      this.numberOfDaysInAGivenClassroom = numberOfDaysInAGivenClassroom;
29    }
30  
31    @Override
32    public boolean equals(Object o) {
33      if (this == o) {
34        return true;
35      }
36      if (o == null || getClass() != o.getClass()) {
37        return false;
38      }
39      substituteTeacher that = (substituteTeacher) o;
40      return Objects
41          .equals(numberOfDaysInAGivenClassroom, that.numberOfDaysInAGivenClassroom);
42    }
43  
44    @Override
45    public int hashCode() {
46      return Objects.hash(numberOfDaysInAGivenClassroom);
47    }
48  
49    @Override
50    public String toString() {
51      return "substituteTeacher{" +
52          ", name=" + name +
53          ", classroom='" + classroom + '\'' +
54          ", coteacher=" + coteacher +
55          ", vacationStatus=" + vacationStatus +
56          ", numberOfDaysInGivenClassroom=" + numberOfDaysInGivenClassroom +
57          '}';
58    }
59  }
