6   /**
7    * A Painting class for interior services. The painting class also keeps track of the number of pets
8    * the property has in addition to the invoiceID, property address, property size, a check to see if
9    * the property is a routinely servicing, and a track of the number of services.
10   */
11  public class Painting extends abstractInterior {
12  
13    private Double finalPrice;
14  
15    /**
16     * Constructs a new Painting service with the given parameters.
17     *
18     * @param invoiceID the invoiceID of the painting service
19     * @param propertyAddress the property address of the property
20     * @param propertySize the property size represented as an enum (SMALL, MEDIUM, LARGE)
21     * @param scheduledMonthly a Boolean value checking if the service is monthly or a 1 time service
22     * @param numberOfServices the number of services the property has had
23     * @param numberOfPets the number of pets the property has
24     */
25    public Painting(String invoiceID, String propertyAddress, PropertySizeEnum propertySize,
26        Boolean scheduledMonthly, Integer numberOfServices, Integer numberOfPets) {
27      super(invoiceID, propertyAddress, propertySize,
28          scheduledMonthly, numberOfServices, numberOfPets);
29    }
30  
31    @Override
32    public Double calculatePrice() {
33      Double InteriorPaintingPrice = calculateInteriorPaintingPrice();
34      Double petFee = calculatePetPrice();
35      Double discountPrice = calculateDiscountHelper();
36      Double withoutDiscountPrice = InteriorPaintingPrice + (InteriorPaintingPrice * petFee);
37      return finalPrice = withoutDiscountPrice - (withoutDiscountPrice * discountPrice);
38    }
39  }
