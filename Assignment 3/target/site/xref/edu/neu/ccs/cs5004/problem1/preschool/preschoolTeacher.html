1   package edu.neu.ccs.cs5004.problem1.preschool;
2   
3   import edu.neu.ccs.cs5004.problem1.Name;
4   import edu.neu.ccs.cs5004.problem1.abstractTeacher;
5   import java.util.Objects;
6   
7   /**
8    * A class representing a Preschool Teacher. A Preschool Teacher has all the same parameters of any
9    * other teacher with the addition of having a counting ability to keep track of children who can
10   * read.
11   */
12  public class preschoolTeacher extends abstractTeacher {
13  
14    protected Integer numberOfChildrenAbleToRead;
15    protected preschoolTeacher newPreschoolTeacher;
16  
17    /**
18     * Constructs a new Preschool Teacher with the given parameters.
19     *
20     * @param name the name of the Preschool Teacher
21     * @param coteacher the name of the Co Teacher
22     * @param classroom the classroom of the Preschool Teacher, represented as a String
23     * @param currentNumberOfChildren the current number of children in the classroom
24     * @param maximumNumberOfChildren the maximum number of children in the classroom
25     * @param vacationStatus the vacation status of the teacher, represented as a Boolean
26     * @param numberOfChildrenAbleToRead the current number of children who can read
27     */
28    public preschoolTeacher(Name name, String classroom, Name coteacher,
29        Integer currentNumberOfChildren, Integer maximumNumberOfChildren, Boolean vacationStatus,
30        Integer numberOfChildrenAbleToRead) {
31      super(name, classroom, coteacher, currentNumberOfChildren,
32          maximumNumberOfChildren, vacationStatus);
33      this.numberOfChildrenAbleToRead = numberOfChildrenAbleToRead;
34    }
35  
36    /**
37     * Gets the number of children in the classroom who are able to read.
38     *
39     * @return the number of children who can read
40     */
41    public Integer getNumberOfChildrenAbleToRead() {
42      return numberOfChildrenAbleToRead;
43    }
44  
45    /**
46     * Sets the number of children in the classroom who are able to read.
47     *
48     * @param numberOfChildrenAbleToRead the number of children in the classroom who can read
49     */
50    public void setNumberOfChildrenAbleToRead(Integer numberOfChildrenAbleToRead) {
51      this.numberOfChildrenAbleToRead = numberOfChildrenAbleToRead;
52    }
53  
54    /**
55     * Converts a substitute teacher into a full time preschool teacher by using the abstract class
56     * to help.
57     *
58     * @param substitute the substitute teacher wishing to be converted to full time
59     * @param classroom the classroom she wants to be a full time teacher for
60     * @param coteacher the co-teacher she will be teaching with
61     * @return a new preschool teacher
62     */
63    public abstractTeacher convertSubstituteTeacher(abstractTeacher substitute,
64        String classroom, Name coteacher) {
65      newPreschoolTeacher.setName(substitute.getName());
66      newPreschoolTeacher.setCoteacher(coteacher);
67      newPreschoolTeacher.setClassroom(classroom);
68      return newPreschoolTeacher;
69    }
70  
71    @Override
72    public boolean equals(Object o) {
73      if (this == o) {
74        return true;
75      }
76      if (o == null || getClass() != o.getClass()) {
77        return false;
78      }
79      preschoolTeacher that = (preschoolTeacher) o;
80      return Objects.equals(numberOfChildrenAbleToRead, that.numberOfChildrenAbleToRead);
81    }
82  
83    @Override
84    public int hashCode() {
85      return Objects.hash(numberOfChildrenAbleToRead);
86    }
87  
88    @Override
89    public String toString() {
90      return "preschoolTeacher{" +
91          ", name=" + name +
92          ", coteacher=" + coteacher +
93          ", classroom='" + classroom + '\'' +
94          ", currentNumberOfChildren=" + currentNumberOfChildren +
95          ", maximumNumberOfChildren=" + maximumNumberOfChildren +
96          ", vacationStatus=" + vacationStatus +
97          ", numberOfChildrenAbleToRead=" + numberOfChildrenAbleToRead +
98          '}';
99    }
100 }
